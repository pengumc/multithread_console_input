00073 /*
00074     This function is used to initialze the Maple kernel and return a 
00075     MKernelVector.  This function should not be called more than once.
00076     It should not be executed if Maple is already running (eg via an
00077     external function described by define_external() in Maple), since
00078     external functions provide their own kernel vector. 
00079 
00080     argc:  number of arguments in argv
00081     argv:  argument list (same arguments accepted by the command line
00082                            version of maple; see ?maple for complete list)
00083     cb:  I/O callbacks defined above.  All unused callbacks 
00084          must be set to 0.  If all callbacks are 0, stdout is used
00085          for output, and certain functions like readline() raise errors.
00086     user_data:  pointer to anything; gets passed through to the I/O callbacks
00087     void *info: version information -- always set to NULL
00088     errstr:  preallocated error buffer;  filled in if something goes wrong
00089              at startup.
00090 
00091 */
00092 extern EXT_DECL MKernelVector M_DECL StartMaple( int argc, char *argv[],
00093                 MCallBackVector cb, void *user_data, void *info, char *errstr );

00154 /* Push an error handling routine on the error stack.  When an exception is raised, each
00155  * function on the error handling stack will be called.  The first argument is a
00156  * string containing the exception.  The second argument is the value that was passed
00157  * into MaplePushErrorProc when the function was pushed onto the stack.  This second
00158  * argument can be used to pass data into the error handling function.  
00159  *
00160  * As an example, imagine you have malloc'ed a piece of memory and want to make sure it
00161  * is freed, even if an execption occurs.  You could push an error handling
00162  * function that calls free on its second argument, and then pass the memory block as
00163  * the third argument to MaplePushErrorProc.  If an error does not occur, call
00164  * MaplePopErrorProc and then free the memory block directly */
